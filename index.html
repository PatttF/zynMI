Node.js Integration

Seamlessly integrate Mutated Instruments into your Node.js applications:

// Install the package
npm install mutated-instruments

// Basic usage in Node.js
const MutatedInstruments = require('mutated-instruments');

// Create a synth instance
const synth = new MutatedInstruments.Synth('wavetable');

// Configure parameters
synth.setVolume(0.7);
synth.setAttackTime(0.1);

// Play a note
synth.playNote('C4', 0.5);

// Process audio in real-time
synth.on('audio', (buffer) => {
  processAudio(buffer);
});
